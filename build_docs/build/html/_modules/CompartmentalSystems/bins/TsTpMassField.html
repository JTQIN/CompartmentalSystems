[docs]    def shift(self):
        #pool age and system age increase by one tss
        x,y=self.arr.shape
        ns=(x+1,)+(y+1,)
        new_arr=np.ndarray(ns,dtype=default_data_type())
        new_arr[0,:]=np.zeros(y+1)
        new_arr[:,0]=np.zeros(x+1)
        new_arr[1:,1:]=self.arr
        self.arr=new_arr
